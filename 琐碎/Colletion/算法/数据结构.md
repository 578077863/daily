/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
 */
class LRUCache {
    /**
        规定 链表的前面代表该节点是一段时间内最久没有被使用过,最后是一段时间内最新被使用过
        将数据存放于HashMap中,缓存容量也由map决定,若是超出则去掉链表的最开始结点
        双向链表的作用主要是用于更新每个数据距离最近使用时间的长短
     */
    HashMap<Integer,Node> map;
    DoubleList cache;
    int capacity; 

    //初始化
    public LRUCache(int capacity) {
        map=new HashMap<Integer,Node>();
        cache=new DoubleList();
        this.capacity=capacity;
    }

    public void deleteKey(int key){
        Node node=map.get(key);
        map.remove(key);
        cache.remove(node);
    }
    
    
    /**
        如果该元素存在,就返回 并且更新其在链表中的位置到链尾
        不存在就返回-1
     */
    public int get(int key) {
        if(map.containsKey(key)){
            makeRecently(key);
            return map.get(key).val;// 返回的是value值
        }
        return -1;
    }
    
    public void makeRecently(int key){
        Node node=map.get(key);
        cache.remove(node);
        cache.addLast(node);
    }

    public void put(int key, int value) {
        if(map.containsKey(key)){
            deleteKey(key);
            addRecently(key,value);
            return;
        }

        int size=cache.getSize();
        if(size == capacity){
            removeLeastRecently();
        }
        addRecently(key,value);

    }

    public void addRecently(int key,int value){
        Node node=new Node(key,value);
        map.put(key,node);
        cache.addLast(node);
    }

    public void removeLeastRecently(){
        Node node=cache.removeHead();
        // map.remove(node.val);
        map.remove(node.key);

    }

}


//创建双链表
class DoubleList{
    Node head,tail;
    int size;
    public DoubleList(){
        head=new Node(0,0);
        tail=new Node(0,0);
        head.next=tail;
        tail.pre=head;
        size=0;
    }

    public Node removeHead(){
        if(head.next == tail){
            return null;
        }
        Node node=head.next;
        remove(node);
        return node;
    }
    public void remove(Node node){
        
        node.pre.next=node.next;
        node.next.pre=node.pre;
        size--;
    }

    public void addLast(Node node){
        node.pre=tail.pre;
        tail.pre.next=node;
        node.next=tail;
        tail.pre=node;
        size++;
    }

    public int getSize(){
        return size;
    }
}

//创建结点
class Node{
    int key,val;
    Node pre,next;
    public Node(int key,int val){
        this.key=key;
        this.val=val;
    }
}
